180
181 if (working.size() == 0) return;
182
183 for (i = 0; i < working.size(); i++){
184 if (working[i]->status != PACKAGE_STATE_DRAM_FETCH && working[i]->status != PACKAGE_STATE_DRAM_READY){
185 if (this->channels[get_channel (working[i]->memory_address)].addRequest (working[i])) {
186 working[i]->ram_cycle = orcs_engine.get_global_cycle();
187 working[i]->updatePackageDRAMFetch (0);
188 }
189 } else if (working[i]->status == PACKAGE_STATE_DRAM_READY && working[i]->readyAt <= orcs_engine.get_global_cycle()){
190 wait_time = (orcs_engine.get_global_cycle() - working[i]->ram_cycle);
191 #if MEMORY_DEBUG
192 ORCS_PRINTF ("[MEMC] %lu %lu %s finishes at main memory! Took %lu cycles.\n", orcs_engine.get_global_cycle(), working[i]->memory_address, get_enum_memory_operation_char (working[i]->memory_operation), wait_time)
193 #endif
194 working[i]->updatePackageWait (1);
195 this->total_operations[working[i]->memory_operation]++;
196 if (wait_time < this->min_wait_operations[working[i]->memory_operation]) this->min_wait_operations[working[i]->memory_operation] = wait_time;
197 if (wait_time > this->min_wait_operations[working[i]->memory_operation]) this->max_wait_operations[working[i]->memory_operation] = wait_time;
198 this->total_latency[working[i]->memory_operation] += wait_time;
199 working.erase(std::remove(working.begin(), working.end(), working[i]), working.end());
200 working.shrink_to_fit();
201 }

222 // =======================================================
223 this->controller_bits_shift = 0;
224 this->colbyte_bits_shift = 0;
225 this->colrow_bits_shift = utils_t::get_power_of_two(this->LINE_SIZE);
226 this->channel_bits_shift = utils_t::get_power_of_two(this->BANK_ROW_BUFFER_SIZE);
227 this->bank_bits_shift = this->channel_bits_shift + utils_t::get_power_of_two(this->CHANNEL);
228 this->row_bits_shift = this->bank_bits_shift + utils_t::get_power_of_two(this->BANK);
229
231 for (i = 0; i < utils_t::get_power_of_two(this->LINE_SIZE); i++) {
232 this->colbyte_bits_mask |= 1 << (i + this->colbyte_bits_shift);
233 }
234
236 for (i = 0; i < utils_t::get_power_of_two(this->BANK_ROW_BUFFER_SIZE / this->LINE_SIZE); i++) {
237 this->colrow_bits_mask |= 1 << (i + this->colrow_bits_shift);
238 }
239
240 this->not_column_bits_mask = ~(colbyte_bits_mask | colrow_bits_mask);
241
243 for (i = 0; i < utils_t::get_power_of_two(this->CHANNEL); i++) this->channel_bits_mask |= 1 << (i + channel_bits_shift);
244
246 for (i = 0; i < utils_t::get_power_of_two(this->BANK); i++) this->bank_bits_mask |= 1 << (i + bank_bits_shift);
247
249 for (i = row_bits_shift; i < utils_t::get_power_of_two((uint64_t)INT64_MAX+1); i++) {
250 this->row_bits_mask |= 1 << i;
