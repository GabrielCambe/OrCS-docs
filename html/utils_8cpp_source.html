48uint64_t utils_t::get_power_of_two(uint64_t n) {
49 if (n == 0) {
50 return 0;
51 }
52
53 ERROR_ASSERT_PRINTF(check_if_power_of_two(n), "Trying to get a log2 of a non power of two number %" PRIu64 ".\n", n);
54 int64_t log2 = -1;
55 while (n != 0) {
56 n >>= 1;
57 log2++;
58 }
59 return log2;
60}

63uint32_t utils_t::check_if_power_of_two(uint64_t n) {
64 uint32_t i, count = 0;
65 uint64_t pow;
67 for (i = 0; i < 64 && n != 0 ; i++) {
68 pow = n & (1 << i);
69 count += (pow != 0);
70 n -= pow;
71 }
72
73 if (count == 1) {
74 return OK;
75 }
76 else {
77 return FAIL;
78 }
79}

81uint64_t utils_t::hash_function(hash_function_t type, uint64_t input1, uint64_t input2, uint64_t bit_size) {
82 uint32_t i;
83 uint64_t bit_size_mask = 0;
84 uint64_t input1_bit_mask = 0;
85 uint64_t input2_bit_mask = 0;
86 uint64_t output = 0;
87
88 for (i = 0; i < bit_size; i++) {
89 bit_size_mask |= 1 << i;
90 }
91
92 switch (type) {
93 case HASH_FUNCTION_XOR_SIMPLE:
94 output = input1 ^ input2;
95 output &= bit_size_mask;
96 break;
97
98 case HASH_FUNCTION_INPUT1_ONLY:
99 output = input1 & bit_size_mask;
100 break;
101
102 case HASH_FUNCTION_INPUT2_ONLY:
103 output = input2 & bit_size_mask;
104 break;
105
106 case HASH_FUNCTION_INPUT1_2BITS:
107 ERROR_ASSERT_PRINTF(bit_size >= 2, "Bit size is smaller than the number of bits from input1.\n")
108 for (i = 0; i < 2; i++) {
109 input1_bit_mask |= (1 << i);
110 }
111 input1 &= input1_bit_mask;
112 input1 = input1 << (bit_size - 2);
113
114 for (i = 0; i < bit_size - 2; i++) {
115 input2_bit_mask |= (1 << i);
116 }
117 input2 &= input2_bit_mask;
118
120 output = input1 | input2;
121 break;
122
123 case HASH_FUNCTION_INPUT1_4BITS:
124 ERROR_ASSERT_PRINTF(bit_size >= 4, "Bit size is smaller than the number of bits from input1.\n")
125 for (i = 0; i < 4; i++) {
126 input1_bit_mask |= (1 << i);
127 }
128 input1 &= input1_bit_mask;
129 input1 = input1 << (bit_size - 4);
130
131 for (i = 0; i < bit_size - 4; i++) {
132 input2_bit_mask |= (1 << i);
133 }
134 input2 &= input2_bit_mask;
135
137 output = input1 | input2;
138
139 break;
140
141 case HASH_FUNCTION_INPUT1_8BITS:
142 ERROR_ASSERT_PRINTF(bit_size >= 8, "Bit size is smaller than the number of bits from input1.\n")
143 for (i = 0; i < 8; i++) {
144 input1_bit_mask |= (1 << i);
145 }
146 input1 &= input1_bit_mask;
147 input1 = input1 << (bit_size - 8);
148
149 for (i = 0; i < bit_size - 8; i++) {
150 input2_bit_mask |= (1 << i);
151 }
152 input2 &= input2_bit_mask;
153
155 output = input1 | input2;
156 break;
157
158 case HASH_FUNCTION_INPUT1_16BITS:
159 ERROR_ASSERT_PRINTF(bit_size >= 16, "Bit size is smaller than the number of bits from input1.\n")
160 for (i = 0; i < 16; i++) {
161 input1_bit_mask |= (1 << i);
162 }
163 input1 &= input1_bit_mask;
164 input1 = input1 << (bit_size - 16);
165
166 for (i = 0; i < bit_size - 16; i++) {
167 input2_bit_mask |= (1 << i);
168 }
169 input2 &= input2_bit_mask;
170
172 output = input1 | input2;
173 break;
174
175 case HASH_FUNCTION_INPUT1_32BITS:
176 ERROR_ASSERT_PRINTF(bit_size >= 32, "Bit size is smaller than the number of bits from input1.\n")
177 for (i = 0; i < 32; i++) {
178 input1_bit_mask |= (1 << i);
179 }
180 input1 &= input1_bit_mask;
181 input1 = input1 << (bit_size - 32);
182
183 for (i = 0; i < bit_size - 32; i++) {
184 input2_bit_mask |= (1 << i);
185 }
186 input2 &= input2_bit_mask;
187
189 output = input1 | input2;
190 break;
191 }
192 return output;
193}

237std::string utils_t::address_to_binary(uint64_t address) {
239 std::string answer;
240 answer.clear();
241
242 uint64_t z;
243 uint32_t count = 0;
244
245 for (z = (uint64_t)INT64_MAX+1; z > 0; z >>= 1) {
246 if ((address & z) == z) {
247 answer = answer + "1";
248 }
249 else {
250 answer = answer + "0";
251 }
252 count++;
253 if (count%4 == 0) {
254 answer = answer + " ";
255 }
256 }
257 return answer;
258}

261std::string utils_t::print_mask_of_bits(uint32_t line_size, uint32_t line_number, uint32_t assoc) {
262 uint32_t i, offset = 0, index = 0;
263
265 static char a[65];
266 a[0] = '\0';
267
268 offset = get_power_of_two(line_size);
269 index = get_power_of_two(line_number/assoc) + offset;
270 for (i = 0 ; i < 64 ; i++) {
271 if (i < 32) {
272 strcat(a , ".");
273 }
274 else if (i < 64-index) {
275 strcat(a , "T");
276 }
277 else if (i < 64-offset) {
278 strcat(a , "I");
279 }
280 else {
281 strcat(a , "O");
282 }
283 }
284
285 return a;
286}

514void utils_t::process_mem_usage(double *vm_usage, double *resident_set) {
515 using std::ios_base;
516 using std::ifstream;
517 using std::string;
518
519 *vm_usage = 0.0;
520 *resident_set = 0.0;
521
523 ifstream stat_stream("/proc/self/stat", ios_base::in);
524
526 string pid, comm, state, ppid, pgrp, session, tty_nr;
527 string tpgid, flags, minflt, cminflt, majflt, cmajflt;
528 string utime, stime, cutime, cstime, priority, nice;
529 string O, itrealvalue, starttime;
530
532 uint64_t vsize;
533 uint64_t rss;
534
535 stat_stream >> pid >> comm >> state >> ppid >> pgrp >> session >> tty_nr
536 >> tpgid >> flags >> minflt >> cminflt >> majflt >> cmajflt
537 >> utime >> stime >> cutime >> cstime >> priority >> nice
538 >> O >> itrealvalue >> starttime >> vsize >> rss;
539
540 stat_stream.close();
541
542 double page_size_mb = sysconf(_SC_PAGE_SIZE) / 1024.0 / 1024.0;
543 *vm_usage = vsize / 1024.0 / 1024.0;
544 *resident_set = rss * page_size_mb;
545}
