850 // uint32_t position;
851 // Trace ->fetchBuffer
852 for (uint32_t i = 0; i < FETCH_WIDTH; i++) {
853 if (orcs_engine.cacheManager->available (this->processor_id, MEMORY_OPERATION_INST)){
854 operation.package_clean();
855 //bool updated = false;
856 //=============================
857 //Stall full fetch buffer
858 //=============================
859 if (this->fetchBuffer.is_full()) {
860 this->add_stall_full_FetchBuffer();
861 break;
862 }
863 //=============================
864 //Stall branch wrong predict
865 //=============================
866 if (this->get_stall_wrong_branch() > orcs_engine.get_global_cycle()){
867 break;
868 }
869 //=============================
870 //Get new Opcode
871 //=============================

873 this->traceIsOver = true;
874 break;

876 #if FETCH_DEBUG

880 //add control variables
881 //============================

883 operation.readyAt = orcs_engine.get_global_cycle() + FETCH_LATENCY;
884 this->fetchCounter++;
885
886 //============================
888 //============================
889
890 if (this->hasBranch){
891 //solve

908 this->hasBranch = true;
909 }
910 //============================
911 //Insert into fetch buffer
912 //============================
913 if (POSITION_FAIL == this->fetchBuffer.push_back(operation)){
914 break;
915 }
916
917 #if PROCESSOR_DEBUG

945 }
946 }

950 ===========================
951 Elimina os elementos do fetch buffer
952 ============================================================================
953 Divide the opcode into
954 1st. uop READ MEM. + unaligned
955 2st. uop READ 2 MEM. + unaligned
956 3rd. uop BRANCH
957 4th. uop ALU
958 5th. uop WRITE MEM. + unaligned
959 ============================================================================
960 To maintain the right dependencies between the uops and opcodes
961 If the opcode generates multiple uops, they must be in this format:
962
963 READ ReadRegs = BaseRegs + IndexRegs
964 WriteRegs = 258 (Aux Register)
965
966 ALU ReadRegs = * + 258 (Aux Register) (if is_read)
967 WriteRegs = * + 258 (Aux Register) (if is_write)
968
969 WRITE ReadRegs = * + 258 (Aux Register)
970 WriteRegs = NULL
971 ============================================================================
972*/
973void processor_t::decode(){
974 #if DECODE_DEBUG
975 ORCS_PRINTF("Decode Stage\n")
976 if(orcs_engine.get_global_cycle()>WAIT_CYCLE){
977 ORCS_PRINTF("Opcode to decode %s\n", this->fetchBuffer.front()->content_to_string2().c_str())
978 }
979 #endif
980 uop_package_t new_uop;
981 int32_t statusInsert = POSITION_FAIL;
982 for (size_t i = 0; i < DECODE_WIDTH; i++)
983 {
984 if (this->fetchBuffer.is_empty() ||
985 this->fetchBuffer.front()->status != PACKAGE_STATE_READY ||
986 this->fetchBuffer.front()->readyAt > orcs_engine.get_global_cycle())
987 {
988 //if (!this->fetchBuffer.is_empty()) ORCS_PRINTF ("NOT READY %s %u %lu\n", get_enum_package_state_char (this->fetchBuffer.front()->status), this->fetchBuffer.front()->readyAt, orcs_engine.get_global_cycle())
989 break;
990 }
991 if (this->decodeBuffer.get_capacity() - this->decodeBuffer.get_size() < MAX_UOP_DECODED)

993 this->add_stall_full_DecodeBuffer();
994 //ORCS_PRINTF ("DECODE FULL %s %s %u %lu\n", get_enum_package_state_char (this->fetchBuffer.front()->status), get_enum_instruction_operation_char (this->fetchBuffer.front()->opcode_operation), this->fetchBuffer.front()->readyAt, orcs_engine.get_global_cycle())
995 break;
996 }
997 ERROR_ASSERT_PRINTF(this->decodeCounter == this->fetchBuffer.front()->opcode_number, "Trying decode out-of-order");
998 this->decodeCounter++;
999
1000 //HIVE
1001 if (get_HAS_HIVE()){
1002 if (this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_FP_ALU ||
1003 this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_FP_DIV ||
1004 this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_FP_MUL ||
1005 this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_INT_ALU ||
1006 this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_INT_DIV ||
1007 this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_INT_MUL ||
1008 this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_LOCK ||
1009 this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_UNLOCK){
1010 new_uop.package_clean();
1011 new_uop.opcode_to_uop (this->uopCounter++,
1012 this->fetchBuffer.front()->opcode_operation,
1013 0,
1014 1,
1015 *this->fetchBuffer.front());
1016
1017 new_uop.is_hive = true;
1018 new_uop.is_vima = false;
1019 new_uop.hive_read1 = this->fetchBuffer.front()->hive_read1;
1020 new_uop.hive_read2 = this->fetchBuffer.front()->hive_read2;
1021 new_uop.hive_write = this->fetchBuffer.front()->hive_write;
1022
1023 new_uop.updatePackageWait (DECODE_LATENCY);
1024 new_uop.born_cycle = orcs_engine.get_global_cycle();
1025 this->total_operations[new_uop.opcode_operation]++;
1026 statusInsert = this->decodeBuffer.push_back(new_uop);
1027 #if DECODE_DEBUG
1028 ORCS_PRINTF("uop created %s\n", this->decodeBuffer.back()->content_to_string2().c_str())
1029 #endif
1030 ERROR_ASSERT_PRINTF(statusInsert != POSITION_FAIL, "Erro, Tentando decodificar mais uops que o maximo permitido")
1031 this->fetchBuffer.pop_front();
1032 return;
1033 } else if (this->fetchBuffer.front()->opcode_operation == INSTRUCTION_OPERATION_HIVE_LOAD){
1034 new_uop.package_clean();
1035 new_uop.opcode_to_uop (this->uopCounter++,
1036 this->fetchBuffer.front()->opcode_operation,
1037 this->fetchBuffer.front()->read_address,
1038 this->fetchBuffer.front()->read_size,
1039 *this->fetchBuffer.front());
1040
1041 new_uop.is_hive = true;
1042 new_uop.is_vima = false;
1043 new_uop.hive_read1 = this->fetchBuffer.front()->hive_read1;
1044 new_uop.read_address = this->fetchBuffer.front()->read_address;
1045 new_uop.hive_read2 = this->fetchBuffer.front()->hive_read2;
1046 new_uop.read2_address = this->fetchBuffer.front()->read2_address;
1047 new_uop.hive_write = this->fetchBuffer.front()->hive_write;
1048 new_uop.write_address = this->fetchBuffer.front()->write_address;
1049
1050 new_uop.updatePackageWait (DECODE_LATENCY);
1051 new_uop.born_cycle = orcs_engine.get_global_cycle();
1052 this->total_operations[new_uop.opcode_operation]++;
1053 statusInsert = this->decodeBuffer.push_back(new_uop);
1054 #if DECODE_DEBUG
