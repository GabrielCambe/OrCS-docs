361bool trace_reader_t::trace_next_dynamic(uint32_t *next_bbl) {
362 static char file_line[TRACE_LINE_SIZE];
363 file_line[0] = '\0';
364
365 bool valid_dynamic = false;
366 *next_bbl = 0;
367

370 if (gzeof(this->gzDynamicTraceFile)) {
371 return FAIL;
372 }
373 char *buffer = gzgets(this->gzDynamicTraceFile, file_line, TRACE_LINE_SIZE);
374 if (buffer == NULL) {
375 return FAIL;
376 }
377
379 if (file_line[0] == '\0' || file_line[0] == '#') {
380 DEBUG_PRINTF("Dynamic trace line (empty/comment): %s\n", file_line);

384 DEBUG_PRINTF("Dynamic trace line (synchronization): %s\n", file_line);

391
