88template <class CB_TYPE>
89void circular_buffer_t<CB_TYPE>::allocate(uint32_t elements) {
90 this->capacity = elements;
91 this->data = new CB_TYPE[this->capacity];
92
93 ERROR_ASSERT_PRINTF(this->data != NULL, "Could not allocate the circular buffer size.\n")

96// ============================================================================
97template <class CB_TYPE>
98inline CB_TYPE& circular_buffer_t<CB_TYPE>::operator[](uint32_t index) {
99 ERROR_ASSERT_PRINTF(index < this->capacity, "Trying to access beyond the circular buffer size.\n")
100 ERROR_ASSERT_PRINTF(this->data != NULL, "Trying to access beyond the circular buffer size.\n")
101
102 uint32_t position = this->beg_index + index;
103 if (position >= this->capacity)
104 position -= this->capacity;
105
106 return this->data[position];
107}

133// ============================================================================
135template <class CB_TYPE>
136int32_t circular_buffer_t<CB_TYPE>::push_back(const CB_TYPE& new_element) {
137 ERROR_ASSERT_PRINTF(this->data != NULL, "Trying to access beyond the circular buffer size.\n")
138
139 int32_t virtual_position = POSITION_FAIL;
140
141 if (!is_full()) {
142 virtual_position = this->size;
143 this->size++;
144 this->data[end_index] = new_element;
145 this->end_index++;
146 if (this->end_index >= this->capacity)
147 this->end_index = 0;
148 }
149
150 return virtual_position;
151}

166// ============================================================================
169template <class CB_TYPE>
170inline CB_TYPE* circular_buffer_t<CB_TYPE>::back() {
171 if (this->size == 0) {
172 return NULL;
173 }
174 else {
175 uint32_t position = this->beg_index + this->size - 1;
176 if (position >= this->capacity)
177 position -= this->capacity;
178
179 return &this->data[position];
180 }
181}

183// ============================================================================
185template <class CB_TYPE>
186void circular_buffer_t<CB_TYPE>::pop_front() {
187 if (this->size > 0) {
188 this->size--;
189 this->data[beg_index].package_clean();
190
191 this->beg_index++;
192 if (this->beg_index >= this->capacity)
193 this->beg_index = 0;
194 }
195}
196
197// ============================================================================
199template <class CB_TYPE>
200void circular_buffer_t<CB_TYPE>::pop_push() {
201 CB_TYPE older = this->data[beg_index];
202 this->data[beg_index].package_clean();
203
204 this->beg_index++;
205 if (this->beg_index >= this->capacity)
206 this->beg_index = 0;
207
208 this->data[end_index] = older;
209
210 this->end_index++;
211 if (this->end_index >= this->capacity)
212 this->end_index = 0;
213}
