47std::string reorder_buffer_line_t::content_to_string() {
48 std::string content_string;
49 content_string = "";
50
51 #ifndef SHOW_FREE_PACKAGE
52 if (this->uop.status == PACKAGE_STATE_FREE) {
53 return content_string;
54 }
55 #endif
56
57 content_string = this->uop.content_to_string();
58 content_string = content_string + " | Stage:" + get_enum_processor_stage_char(this->stage);
59 content_string = content_string + " | Reg.Wait:" + utils_t::uint32_to_string(this->wait_reg_deps_number);
60 content_string = content_string + " | WakeUp:" + utils_t::uint32_to_string(this->wake_up_elements_counter);
61 content_string = content_string + " | ReadyAt: " + utils_t::uint64_to_string(this->uop.readyAt);
62 content_string = content_string + " | Sent: " + utils_t::bool_to_string(this->sent);
63 if(this->mob_ptr != NULL){
64 content_string = content_string + this->mob_ptr->content_to_string();
65 }
66 return content_string;
67}
